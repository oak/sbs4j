231     @Test
232     void dataWithWrongInputLength_should_throw_BufferUnderflowException() {
233         DeserializerBuffer deserializerBuffer1 = new DeserializerBuffer("");
234 
235         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readBool);
236         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readU32);
237         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readU64);
238         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readU128);
239         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readU256);
240         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readU512);
241         assertThrows(ValueDeserializationException.class, deserializerBuffer1::readString);
242         assertThrows(ValueDeserializationException.class, () -> deserializerBuffer1.readByteArray(32));
243 
244         DeserializerBuffer deserializerBuffer2 = new DeserializerBuffer("01");
245 
246         assertThrows(ValueDeserializationException.class, deserializerBuffer2::readU512);
247 
248         DeserializerBuffer deserializerBuffer3 = new DeserializerBuffer("01");
249 
250         assertThrows(ValueDeserializationException.class, deserializerBuffer3::readString);
