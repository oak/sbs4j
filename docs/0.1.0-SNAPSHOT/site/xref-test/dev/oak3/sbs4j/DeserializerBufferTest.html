50      @Test
51      void validateDeserialize_with_SampleData() throws ValueDeserializationException {
52          for (TestData<?> testData : TestData.SUCCESS_TEST_DATA) {
53              DeserializerBuffer deserializerBuffer = new DeserializerBuffer(testData.getByteValue());
54              switch (testData.getName()) {
55                  case TestData.BOOLEAN:
56                      boolean valueBool = deserializerBuffer.readBool();
57                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueBool);
58                      assertEquals(testData.getValue(), valueBool);
59                      break;
60                  case TestData.BYTE_ARRAY:
61                      byte[] valueByteArray = deserializerBuffer.readByteArray(testData.getByteValue().length);
62                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueByteArray);
63                      assertArrayEquals((byte[]) testData.getValue(), valueByteArray);
64                      break;
65                  case TestData.U_8:
66                      byte valueU8 = deserializerBuffer.readU8();
67                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueU8);
68                      assertEquals(testData.getValue(), valueU8);
69                      break;
70                  case TestData.U_32:
71                      long valueU32 = deserializerBuffer.readU32();
72                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueU32);
73                      assertEquals(testData.getValue(), valueU32);
74                      break;
75                  case TestData.F_32:
76                      float valueF32 = deserializerBuffer.readF32();
77                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueF32);
78                      assertEquals(testData.getValue(), valueF32);
79                      break;
80                  case TestData.F_64:
81                      double valueF64 = deserializerBuffer.readF64();
82                      LOGGER.debug("Expected: {}, Actual: {}", testData.getValue(), valueF64);

85                  case TestData.U_64:

124     @Test
125     void dataWithWrongInputLength_should_throw_BufferUnderflowException() throws InvalidByteStringException {
126         DeserializerBuffer deserializerBuffer1 = new DeserializerBuffer("");
127 
128         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readBool);
129         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readU32);
130         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readU64);
131         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readU128);
132         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readU256);
133         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readU512);
134         assertThrows(BufferUnderflowException.class, deserializerBuffer1::readString);
135         assertThrows(BufferUnderflowException.class, () -> deserializerBuffer1.readByteArray(32));
136 
137         DeserializerBuffer deserializerBuffer2 = new DeserializerBuffer("01");
138 
139         assertThrows(BufferUnderflowException.class, deserializerBuffer2::readU512);
140 
141         DeserializerBuffer deserializerBuffer3 = new DeserializerBuffer("01");
142 
143         assertThrows(BufferUnderflowException.class, deserializerBuffer3::readString);
144     }
